    run VARCHAR(255),
    chr VARCHAR(50),
    pos BIGINT,
    ref VARCHAR(255),
    alt VARCHAR(255),
    dp BIGINT,
    g_ad_2 BIGINT,
    effect VARCHAR(255),
    funclass VARCHAR(255),
    codon VARCHAR(255),
    variation VARCHAR(255),
    aa_len BIGINT,
    protein_name VARCHAR(255),
    ref_aa VARCHAR(255),
    alt_aa VARCHAR(255),
    protein_position BIGINT,
    ver VARCHAR(50),
    updated TIMESTAMP
);
